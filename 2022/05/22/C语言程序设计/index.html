1. ```c++
   #include <stdio.h>
   void main() {
   //    编程序。输入单精度型一维数组a[10]，计算并输出a数组中所有元素的平均值。
       int arr[10] = {0}, sum = 0;
       for (int i = 0; i < 10; i++) {
           scanf("%d", &arr[i]);
       }
       for (int i = 0; i < 10; i++) {
           sum += arr[i];
       }
       float sum1 = 0;
       sum1 = sum / 10.0;
       printf("%.3f", sum1);
   }
   

3. ```c++
   #include <stdio.h>
   void main() {
       /* 算法： 数组a[3][3]    求一个3×3矩阵对角线元素之和。
         a[0][0]    a[0][1]   a[0][2]
         a[1][0]    a[1][1]   a[1][2]
         a[2][0]    a[2][1]   a[2][2]  */
   
       float arr[3][3] = {0};
       for (int i = 0; i < 3; i++) {
           for (int j = 0; j < 10; j++) {
               scanf("%f", &arr[i][j]);
           }
       }
       float sum = 0;
       for (int i = 0; i < 10; i++) {
           for (int j = 0; j < 10; j++) {
               if (i == j || (i + j == 2)) {
                   sum += arr[i][j];
               }
           }
       }
   }
